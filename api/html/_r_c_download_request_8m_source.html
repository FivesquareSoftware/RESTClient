00061 - (void)connection:(NSURLConnection *)connection didReceiveResponse:(NSURLResponse *)response {
00062         [super connection:connection didReceiveResponse:response];
00063 #ifdef BROKEN_DOWNLOAD_DELEGATE
00064         // If the delegate calls are broken (do not actually point to a valid file on download) then we need to create a filename ourselves and use that
00065         NSString *filename = nil;
00066         NSString *contentDisposition = [[(NSHTTPURLResponse *)response allHeaderFields] objectForKey:@"Content-Disposition"];
00067         if (contentDisposition.length) {
00068                 NSRange filenameRange = [contentDisposition rangeOfString:@"filename="];
00069                 if (filenameRange.location != NSNotFound) {
00070                         filename = [contentDisposition substringFromIndex:filenameRange.location+filenameRange.length];
00071                         filename = [filename stringByTrimmingCharactersInSet:[NSCharacterSet characterSetWithCharactersInString:@"\""]];
00072                         self.downloadedFilename = filename;
00073                 }               
00074         }
00075 
00076         CFUUIDRef UUID = CFUUIDCreate(NULL);
00077         NSString *tempFilename = (__bridge_transfer NSString *)CFUUIDCreateString(NULL, UUID);
00078         CFRelease(UUID);
00079 
00080         
00081 
00082         NSString *filePath = [[RESTClient cachesDir] stringByAppendingPathComponent:tempFilename];
00083         
00084         self.downloadedFileTempURL = [NSURL fileURLWithPath:filePath];

00089 - (void)connection:(NSURLConnection *)connection didReceiveData:(NSData *)data {
00090 #ifdef BROKEN_DOWNLOAD_DELEGATE
00091         // When delegate calls are broken we have to write the data to the file ourselves
00092         NSError *writeError = nil;
00093 
00094         NSFileManager *fm = [[NSFileManager alloc] init];
00095         if (NO == [fm fileExistsAtPath:[self.downloadedFileTempURL path]]) {
00096                 if (NO == [data writeToURL:self.downloadedFileTempURL options:NSDataWritingAtomic error:&writeError]) {
00097                         RCLog(@"Could not write to downloaded file URL: %@ (%@)", [writeError localizedDescription],[writeError userInfo]);
00098                         self.error = writeError;
00099                         [self handleConnectionFinished];
00100                 }
00101         } else {
00102                 NSFileHandle *fh = [NSFileHandle fileHandleForUpdatingURL:self.downloadedFileTempURL error:&writeError];
00103                 if (fh) {
00104                         [fh seekToEndOfFile];
00105                         [fh writeData:data];
00106                         [fh closeFile];
00107                 } else {
00108                         RCLog(@"Could not get filehandle for writing to downloaded file URL: %@ (%@)", [writeError localizedDescription],[writeError userInfo]);
00109                         self.error = writeError;
00110                         [self handleConnectionFinished];
00111                 }
00112         }
00113         NSUInteger dataLength = [data length];
00114         self.receivedContentLength += dataLength;
00115         [self handleConnectionDidReceiveData];
