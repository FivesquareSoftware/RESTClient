00237 - (RCResource *) resource:(id)relativePathObject {
00238         NSAssert(URL_ != nil,@"Cannot construct a resource without a base URL!");
00239         
00240         NSString *relativePath;
00241         if ([relativePath isKindOfClass:[NSString class]]) {
00242                 relativePath = relativePathObject;
00243         } else {
00244                 relativePath = [relativePathObject description];
00245         }
00246         
00247         //rdar: 10487909, must remove preceding slash because URLByAppendingPathComponent is adding one incorrectly
00248         relativePath = [relativePath stringByTrimmingCharactersInSet:[NSCharacterSet characterSetWithCharactersInString:@"/"]];
00249         relativePath = [relativePath stringByAddingPercentEscapesUsingEncoding:NSUTF8StringEncoding];
00250         
00251         NSString *queryString = nil;
00252         NSRange queryRange = [relativePath rangeOfString:@"?"];
00253         if (queryRange.location != NSNotFound) {
00254                 queryString = [relativePath substringFromIndex:queryRange.location+1];
00255                 relativePath = [relativePath substringToIndex:queryRange.location];
00256         }
00257         
00258         NSURL *resourceURL = [URL_ URLByAppendingPathComponent:relativePath];
00259         resourceURL = [resourceURL standardizedURL];
00260         
00261         if (queryString.length) {
00262                 resourceURL = [NSURL URLWithString:[NSString stringWithFormat:@"%@?%@",[resourceURL absoluteString],queryString]];
00263         }
00264         
00265         RCResource *resource = [RCResource withURL:[resourceURL absoluteString]];
00266         
00267         resource.parent = self;
00268         
00269         return resource;

00520         dispatch_semaphore_t request_sema = dispatch_semaphore_create(1);
00521     __block RCResponse *localResponse = nil;
00522         
00523         RCCompletionBlock completionBlock = ^(RCResponse *response){
00524                 localResponse = response;
00525                 dispatch_semaphore_signal(request_sema);
00526         };
00527         
00528         RCAbortBlock abortBlock = ^(RCRequest *request) {
00529                 dispatch_semaphore_signal(request_sema);
00530         };
00531 
00532         dispatch_semaphore_wait(request_sema, DISPATCH_TIME_FOREVER);
00533         [self runRequest:request withCompletionBlock:completionBlock abortBlock:abortBlock];
00534 
00535         if ([NSThread currentThread] == [NSThread mainThread]) {
00536                 while(dispatch_semaphore_wait(request_sema, 0.01) != 0) {
00537                         [[NSRunLoop currentRunLoop] runMode:NSDefaultRunLoopMode beforeDate:[NSDate distantPast]];
00538                 }
00539         } else {
00540                 dispatch_semaphore_wait(request_sema, DISPATCH_TIME_FOREVER);
00541         }
00542         dispatch_semaphore_signal(request_sema);
00543         dispatch_release(request_sema);
00544     return localResponse;
